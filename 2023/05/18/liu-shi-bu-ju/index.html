流式布局

FlowLayout

支持属性

  1. 支持FlowLayout设置margin
  2. 支持FlowLayout设置padding
  3. 支持FlowLayout.children设置margin
  4. 考虑了FlowLayout在AT_MOST下的测量宽高

效果图

流程

  1. LayoutInflater.generateLayoutParams(attrs: AttributeSet?)

    提供MarginLayoutParams，使children的margin生效

  2. onMeasure()

    测量所有child，考虑FlowLayout本身的padding，以及child的margin

  3. onLayout()

    根据测量所得的参数进行子View的布局

代码

package com.guru.composecookbook

import android.content.Context
import android.util.AttributeSet
import android.util.Log
import android.view.View
import android.view.ViewGroup
import androidx.core.view.marginBottom
import androidx.core.view.marginLeft
import androidx.core.view.marginRight
import androidx.core.view.marginTop
import java.lang.Integer.max

/**
 * 1. onMeasure()
 *      --> FlowLayout的padding, children的margin
 *      --> SpecMode
 *          --> EXACTLY
 *          --> AtMost
 * 2. onLayout()
 *      --> FlowLayout的padding, children的margin
 */
class FlowLayout @JvmOverloads constructor(
    context: Context?,
    attrs: AttributeSet? = null,
    defStyleAttr: Int = 0,
    defStyleRes: Int = 0
) : ViewGroup(context, attrs, defStyleAttr, defStyleRes) {

    private lateinit var views: MutableList<List<View>>
    private lateinit var heights: MutableList<Int>
    private lateinit var lineViews: MutableList<View>

    override fun generateLayoutParams(attrs: AttributeSet?): LayoutParams {
        return MarginLayoutParams(context, attrs)
    }

    override fun onMeasure(widthMeasureSpec: Int, heightMeasureSpec: Int) {
        measureFlowChildren(widthMeasureSpec, heightMeasureSpec)
    }

    private fun measureFlowChildren(widthMeasureSpec: Int, heightMeasureSpec: Int) {
        views = mutableListOf()
        heights = mutableListOf()
        lineViews = mutableListOf()

        //父项给到this的宽及Mode
        val widthSpecMode = MeasureSpec.getMode(widthMeasureSpec)
        val widthSpecSize = MeasureSpec.getSize(widthMeasureSpec)
        //父项给到this的高及Mode
        val heightSpecMode = MeasureSpec.getMode(heightMeasureSpec)
        val heightSpecSize = MeasureSpec.getSize(heightMeasureSpec)

        //测量宽高,this的总高，以及line的最大宽度
        var selfMaxWidth = 0
        var selfHeight = 0
        //当前行
        var curWidth = 0
        var curHeight = 0

        val update = {
            views.add(lineViews)
            heights.add(curHeight)

            //更新this的宽高
            selfHeight += curHeight
            selfMaxWidth = max(selfMaxWidth, curWidth)
        }

        (0 until childCount).forEach { index ->
            //先测量child
            val child = getChildAt(index)
            child.measure(widthSpecSize, heightSpecSize)
            val layoutParams = child.layoutParams as MarginLayoutParams

            Log.v(
                "child",
                "child{marginLeft:${child.marginLeft},marginTop:${child.marginTop},marginRight:${child.marginRight},marginBottom:${child.marginBottom}}, layoutParams.marginStart = ${layoutParams.marginStart}"
            )
            //拿到child的测量宽高
            val innerLineWidth = curWidth + child.measuredWidth + child.margin()
            val lineWidth = innerLineWidth + paddingLeft + paddingRight
            if (lineWidth > widthSpecSize) {
                update()

                //重置
                lineViews = mutableListOf()
                curWidth = 0
                curHeight = 0
            }

            //当前 child 的尺寸大于 widthSpecSize
            if (child.measuredWidth + child.margin() + horizontalPadding > widthSpecSize) {
                views.add(listOf(child))
                val viewHeight = child.measuredHeight + child.margin(true)
                heights.add(viewHeight)

                //更新this的宽高
                selfHeight += viewHeight
                selfMaxWidth = widthSpecSize
            } else {
                curWidth += child.measuredWidth + child.margin()
                curHeight = max(curHeight, child.measuredHeight + child.margin(true))
                lineViews.add(child)
            }
        }

        if (lineViews.isNotEmpty()) {
            update()
        }

        var width = selfMaxWidth
        var height = selfHeight

        if (widthSpecMode == MeasureSpec.EXACTLY) {
            width = widthSpecSize
        }
        if (heightSpecMode == MeasureSpec.EXACTLY) {
            height = heightSpecSize
        }
        Log.v(
            DEBUG,
            "MeasureSpec{widthSpecSize = $widthSpecSize, heightSpecSize = $heightSpecSize}"
        )
        Log.v(DEBUG, "FlowLayout{width = $width, height = $height}")
        setMeasuredDimension(width, height)
    }


    /**
     * 以下四个参数是this相对于this.parent的
     * @param l
     * @param t
     * @param r
     * @param b
     */
    override fun onLayout(changed: Boolean, l: Int, t: Int, r: Int, b: Int) {
        measureChildrenLayout(changed, l, t, r, b)
    }

    private fun measureChildrenLayout(
        changed: Boolean,
        left: Int,
        top: Int,
        right: Int,
        bottom: Int
    ) {
        Log.v(DEBUG, "left=$left, top=$top, right=$right, bottom=$bottom")
        var curY = paddingTop
        views.forEachIndexed { index, line ->
            var curX = paddingLeft
            line.forEach { lineChild ->
                val l = curX + lineChild.marginLeft
                val t = curY + lineChild.marginTop
                var r = curX + lineChild.measuredWidth + lineChild.marginLeft
                val b = curY + lineChild.measuredHeight + lineChild.marginTop

                if (lineChild.measuredWidth + lineChild.margin() + horizontalPadding > measuredWidth) {
                    r = measuredWidth - paddingLeft - lineChild.marginLeft
                }

                lineChild.layout(l, t, r, b)
                curX += lineChild.measuredWidth + lineChild.margin()
            }
            curY += heights[index]
        }
    }

    private fun View.margin(isVertical: Boolean = false): Int {
        return if (isVertical) {
            this.marginTop + this.marginBottom
        } else {
            this.marginLeft + this.marginRight
        }
    }

    private inline val ViewGroup.horizontalPadding: Int get() = paddingLeft + paddingRight

    companion object {
        const val DEBUG = "cgf"
    }
}
